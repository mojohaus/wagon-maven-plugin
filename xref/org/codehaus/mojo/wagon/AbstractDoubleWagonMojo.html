22  /**
23   * Provides base functionality for dealing with I/O using single wagon.
24   */
25  public abstract class AbstractDoubleWagonMojo
26      extends AbstractWagonMojo
27  {
28  
29      /**
30       * The URL to the source repository.
31       * 
32       * @parameter property="wagon.source"
33       * @required
34       */
35      protected String source;
36  
37      /**
38       * The URL to the target repository.
39       * 
40       * @parameter property="wagon.target"
41       * @required
42       */
43      protected String target;
44  
45      /**
46       * settings.xml's server id of the source repository. This is used when wagon needs extra authentication
47       * information.
48       * 
49       * @parameter property="wagon.sourceId" default-value="source"
50       */
51      protected String sourceId;
52  
53      /**
54       * settings.xml's server id of the target repository. This is used when wagon needs extra authentication
55       * information.
56       * 
57       * @parameter property="wagon.targetId" default-value="target"
58       */
59      protected String targetId;
60  
61      /**
62       * Optimize the upload by locally compressed all files in one bundle, upload the bundle, and finally remote
63       * uncompress the bundle. This only works with SCP's URL
64       * 
65       * @parameter property="wagon.optimize" default-value="false";
66       */
67  
68      protected boolean optimize = false;
69  
70  }
