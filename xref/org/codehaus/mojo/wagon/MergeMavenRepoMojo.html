22  import java.io.IOException;
23  
24  import org.apache.maven.plugins.annotations.Component;
25  import org.apache.maven.plugins.annotations.Mojo;
26  import org.apache.maven.wagon.Wagon;
27  import org.apache.maven.wagon.WagonException;
28  import org.codehaus.mojo.wagon.shared.MavenRepoMerger;
29  
30  /**
31   * Merge artifacts from one Maven repository to another Maven repository.
32   */
33  @Mojo( name = "merge-maven-repos" , requiresProject = false)
34  public class MergeMavenRepoMojo
35      extends AbstractCopyMojo
36  {
37  
38      @Component
39      private MavenRepoMerger mavenRepoMerger;
40  
41      @Override
42      protected void copy( Wagon srcWagon, Wagon targetWagon )
43          throws IOException, WagonException
44      {
45          mavenRepoMerger.merge( srcWagon, targetWagon, optimize, this.getLog() );
46      }
47  
48  }
