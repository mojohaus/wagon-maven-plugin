10      {
11          WagonFileSet fileSet = new WagonFileSet();
12          fileSet.setDirectory( fromDir );
13  
14          if ( !StringUtils.isBlank( includes ) )
15          {
16              fileSet.setIncludes( StringUtils.split( includes, "," ) );
17          }
18  
19          if ( !StringUtils.isBlank( excludes ) )
20          {
21              fileSet.setExcludes( StringUtils.split( excludes, "," ) );
22          }
23  
24          fileSet.setCaseSensitive( isCaseSensitive );
25
