27      public Wagon334MojoHttpTest(MavenRuntimeBuilder builder )
28          throws Exception
29      {
30          this.maven = builder.withCliOptions( "-B" ).build();
31      }
32  
33      @Test
34      public void testDownload()
35          throws Exception
36      {
37          // issue: https://github.com/mojohaus/wagon-maven-plugin/issues/39
38          File projDir = resources.getBasedir( "http-download-02" );
39          MavenExecution mavenExec = maven.forProject( projDir );
40  
41          MavenExecutionResult result = mavenExec.execute( "clean", "verify" );
42          result.assertErrorFreeLog();
43  
44          File downloadDir = new File(result.getBasedir(), "target/it/http-download/");
45          Assert.assertTrue( new File(downloadDir, "commons-dbutils-1.2-bin.tar.gz.asc" ).exists() );
46          Assert.assertTrue( new File(downloadDir, "commons-dbutils-1.2-bin.tar.gz.asc.md5" ).exists() );
47          Assert.assertTrue( new File(downloadDir, "commons-dbutils-1.2-bin.tar.gz.asc.sha1" ).exists() );
48      }
